```typescript
'use client'

import Link from 'next/link'
import { usePathname } from 'next/navigation'
import { Hash } from 'lucide-react'
import { type Database } from '@/lib/types/database.types'

type Group = Database['public']['Tables']['groups']['Row']

interface GroupLinkProps {
  group: Group
  isSubgroup?: boolean
}

export function GroupLink({ group, isSubgroup = false }: GroupLinkProps) {
  const pathname = usePathname()
  const isActive = pathname === `/${group.id}`

  return (
    <Link
      href={`/${group.id}`}
      className={`
        flex items-center gap-2 rounded-md px-2 py-1.5 text-sm transition-colors duration-150
        ${isSubgroup ? 'ml-4' : ''}
        ${isActive
          ? 'bg-indigo-600/20 text-indigo-300'
          : 'text-white/60 hover:bg-white/5 hover:text-white/90'
        }
      `}
    >
      <Hash
        size={14}
        className={isActive ? 'text-indigo-400' : 'text-white/30'}
        strokeWidth={isActive ? 2 : 1.5}
      />
      <span className="truncate">{group.name}</span>
    </Link>
  )
}
```

```typescript
import { type GroupWithChildren } from '@/lib/actions/groups'
import { GroupLink } from './GroupLink'
import { ChevronRight } from 'lucide-react'

interface GroupTreeProps {
  groups: GroupWithChildren[]
  // ID групп которые уже в папках (чтобы не дублировать)
  groupsInFolders?: Set<string>
}

export function GroupTree({ groups, groupsInFolders = new Set() }: GroupTreeProps) {
  // Показываем только группы НЕ в папках
  const visibleGroups = groups.filter(g => !groupsInFolders.has(g.id))

  if (visibleGroups.length === 0) return null

  return (
    <div className="space-y-0.5">
      {visibleGroups.map(group => (
        <div key={group.id}>
          <GroupLink group={group} />
          {group.children.length > 0 && (
            <div className="mt-0.5 space-y-0.5">
              {group.children.map(child => (
                <GroupLink key={child.id} group={child} isSubgroup />
              ))}
            </div>
          )}
        </div>
      ))}
    </div>
  )
}
```

```typescript
'use client'

import { useState } from 'react'
import { ChevronRight, Folder } from 'lucide-react'
import { type FolderWithItems } from '@/lib/actions/folders'
import { GroupLink } from './GroupLink'

interface FolderListProps {
  folders: FolderWithItems[]
}

function FolderItem({ folder }: { folder: FolderWithItems }) {
  const [isOpen, setIsOpen] = useState(true)

  return (
    <div>
      <button
        onClick={() => setIsOpen(prev => !prev)}
        className="flex w-full items-center gap-1.5 rounded-md px-2 py-1.5 text-xs font-medium uppercase tracking-wide text-white/40 hover:text-white/60 transition-colors duration-150"
      >
        <ChevronRight
          size={12}
          className={`transition-transform duration-150 ${isOpen ? 'rotate-90' : ''}`}
        />
        <Folder size={12} />
        <span className="truncate">{folder.name}</span>
      </button>

      {isOpen && folder.items.length > 0 && (
        <div className="mt-0.5 space-y-0.5">
          {folder.items.map(item => (
            <div key={item.id} className="ml-2">
              <GroupLink group={item.group} />
            </div>
          ))}
        </div>
      )}

      {isOpen && folder.items.length === 0 && (
        <p className="ml-6 py-1 text-xs text-white/20">Пусто</p>
      )}
    </div>
  )
}

export function FolderList({ folders }: FolderListProps) {
  if (folders.length === 0) return null

  return (
    <div className="space-y-1">
      {folders.map(folder => (
        <FolderItem key={folder.id} folder={folder} />
      ))}
    </div>
  )
}
```

Создать src/components/sidebar/Sidebar.tsx (Server Component — загружает данные):

```typescript
import { getGroupsForUser } from '@/lib/actions/groups'
import { getFoldersForUser } from '@/lib/actions/folders'
import { GroupTree } from './GroupTree'
import { FolderList } from './FolderList'
import { MessageSquare } from 'lucide-react'

export async function Sidebar() {
  const [groups, folders] = await Promise.all([
    getGroupsForUser(),
    getFoldersForUser(),
  ])

  // Собираем id групп которые уже в папках
  const groupsInFolders = new Set(
    folders.flatMap(f => f.items.map(item => item.group.id))
  )

  return (
    <aside className="flex h-screen w-72 flex-shrink-0 flex-col bg-[#12121f] border-r border-white/5">
      {/* Заголовок */}
      <div className="flex items-center gap-2.5 px-4 py-4 border-b border-white/5">
        <div className="flex h-7 w-7 items-center justify-center rounded-lg bg-indigo-600/20">
          <MessageSquare size={14} className="text-indigo-400" strokeWidth={1.5} />
        </div>
        <span className="text-sm font-semibold text-white/90 tracking-tight">
          IntraChat
        </span>
      </div>

      {/* Список */}
      <nav className="flex-1 overflow-y-auto py-3 px-2 space-y-4 scrollbar-thin">

        {/* Папки */}
        {folders.length > 0 && (
          <section>
            <FolderList folders={folders} />
          </section>
        )}

        {/* Группы вне папок */}
        {groups.length > 0 && (
          <section>
            {folders.length > 0 && (
              <p className="px-2 mb-1 text-xs font-medium uppercase tracking-wide text-white/25">
                Чаты
              </p>
            )}
            <GroupTree groups={groups} groupsInFolders={groupsInFolders} />
          </section>
        )}

        {/* Пустое состояние */}
        {groups.length === 0 && folders.length === 0 && (
          <div className="px-2 py-4 text-center">
            <p className="text-xs text-white/30">Нет доступных чатов</p>
            <p className="mt-1 text-xs text-white/20">Попросите администратора добавить вас в группу</p>
          </div>
        )}
      </nav>
    </aside>
  )
}
```
